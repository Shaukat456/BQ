## 🧪 Advanced Experimentation Techniques

### Testing Your Creations

```python
# Example of a simple test function
def test_function(function_to_test, test_cases):
    """Run test cases against a function and report results"""
    results = []
    for inputs, expected in test_cases:
        actual = function_to_test(*inputs)
        success = actual == expected
        results.append({
            'inputs': inputs,
            'expected': expected,
            'actual': actual,
            'success': success
        })

    # Calculate success rate
    success_count = sum(r['success'] for r in results)
    rate = success_count / len(results) * 100

    print(f"Success rate: {rate:.2f}% ({success_count}/{len(results)})")

    # Print failed tests
    failed = [r for r in results if not r['success']]
    if failed:
        print("\nFailed tests:")
        for i, test in enumerate(failed):
            print(f"{i+1}. Input: {test['inputs']}")
            print(f"   Expected: {test['expected']}")
            print(f"   Actual: {test['actual']}\n")
```

### Performance Optimization

```python
# Example timing decorator
import time

def measure_time(func):
    """Decorator to measure function execution time"""
    def wrapper(*args, **kwargs):
        start_time = time.time()
        result = func(*args, **kwargs)
        end_time = time.time()
        print(f"{func.__name__} took {end_time - start_time:.6f} seconds to run")
        return result
    return wrapper

# Usage
@measure_time
def my_function():
    # Your code here
    pass
```
